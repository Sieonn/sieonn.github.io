```python
def solution(my_string):
    num = ""
    sol = []

    for v, i in enumerate(my_string):
        if i.isdigit():
            if my_string[v-1].isdigit() and v != 0:
                num = num + i
                if v == len(my_string)-1:
                    sol.append(int(num))
            else:
                num += i
                if v == len(my_string)-1:
                    sol.append(int(num))
        else:
            if num != "":
                sol.append(int(num))
                num = ""
    if sol == []:
        return 0
    return sum(sol)
```
